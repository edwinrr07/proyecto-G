        const chart = new Chart(ctx, {
            type: "line",
            data: {
                labels: [],

                    data: [],
                    borderColor: "blue",
                    borderWidth: 2,

                }
            }
        });

        // 🔹 Escuchar cambios en Firebase
        vibracionRef.on("value", (snapshot) => {
            const vibracion = snapshot.val();
            vibracionActual.textContent = vibracion.toFixed(2);

            const now = new Date().toLocaleTimeString();
            chart.data.labels.push(now);
            chart.data.datasets[0].data.push(vibracion);
           
            if (chart.data.labels.length > 20) { // Limitar puntos en el gráfico
                chart.data.labels.shift();
                chart.data.datasets[0].data.shift();
            }
            chart.update();
        });

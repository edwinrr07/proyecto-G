<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Monitor de Vibraciones</title>
    <script src="https://www.gstatic.com/firebasejs/9.15.0/firebase-app.js"></script>
    <script src="https://www.gstatic.com/firebasejs/9.15.0/firebase-database.js"></script>
    <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>
</head>
<body>
    <h1>Monitor de Vibraciones</h1>
    <h2>Vibraci칩n Actual: <span id="vibracionActual">0</span> m/s</h2>
    <canvas id="grafico"></canvas>

    <script>
        // 游댳 Configurar Firebase
        const firebaseConfig = {
            apiKey: "TU_API_KEY",
            authDomain: "TU_AUTH_DOMAIN",
            databaseURL: "TU_DATABASE_URL",
            projectId: "TU_PROJECT_ID",
            storageBucket: "TU_STORAGE_BUCKET",
            messagingSenderId: "TU_MESSAGING_SENDER_ID",
            appId: "TU_APP_ID"
        };

        firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        const vibracionRef = db.ref("vibracion/actual");
        const vibracionActual = document.getElementById("vibracionActual");

        // 游댳 Configurar gr치fico
        const ctx = document.getElementById("grafico").getContext("2d");
        const chart = new Chart(ctx, {
            type: "line",
            data: {
                labels: [],
                datasets: [{
                    label: "Vibraci칩n (m/s)",
                    data: [],
                    borderColor: "blue",
                    borderWidth: 2,
                    fill: false
                }]
            },
            options: {
                scales: {
                    x: { title: { display: true, text: "Tiempo" } },
                    y: { title: { display: true, text: "Vibraci칩n (m/s)" } }
                }
            }
        });

        // 游댳 Escuchar cambios en Firebase
        vibracionRef.on("value", (snapshot) => {
            const vibracion = snapshot.val();
            vibracionActual.textContent = vibracion.toFixed(2);

            const now = new Date().toLocaleTimeString();
            chart.data.labels.push(now);
            chart.data.datasets[0].data.push(vibracion);
           
            if (chart.data.labels.length > 20) { // Limitar puntos en el gr치fico
                chart.data.labels.shift();
                chart.data.datasets[0].data.shift();
            }
            chart.update();
        });
    </script>
</body>
